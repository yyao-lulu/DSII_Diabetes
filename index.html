diabetes <- read_csv('./data/diabetic_data.csv') %>%
  clean_names()

missing <- vector(length = 50)
for (i in 1:50) {
  diabetes[which(diabetes[,i] == '?'),i] <- NA  # recode '?' to NA.
  missing[i] <- sum(is.na(diabetes[,i]))/dim(diabetes)[1]
}

# missing value proportion
col.mis <- which(missing != 0)
rbind(colnames(diabetes)[col.mis],missing[col.mis])
